import time
start = time.time()  # 시작 시간 저장

ropeCount = int(input())
ropeList = []

for i in range(ropeCount):
  ropeList.append(int(input()))
ropeList.sort()

max = 0;

for i in range(ropeCount):
  nowMax = ropeList[i]*(ropeCount-i)
  if nowMax > max :
    max = nowMax
    
print(max)
