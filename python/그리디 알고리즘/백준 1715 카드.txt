import heapq
import sys


a = int(sys.stdin.readline())

list = []
for i in range(a) :
  
  list.append(int(input()))

result = 0
heapq.heapify(list)

while len(list)>1:
  a = heapq.heappop(list)
  b = heapq.heappop(list)
  heapq.heappush(list, a+b)
  result += (a+b)


print(result)
